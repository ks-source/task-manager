🔍 すべてのノードID確認

コマンド1: すべての data-id を列挙
目的: SVG 内のすべての要素の data-id と data-et を確認
(function() { const svg = document.querySelector('.svg-content-wrapper svg') || document.querySelector('svg'); console.log('=== すべての data-id を持つ要素 ==='); const allElements = svg.querySelectorAll('[data-id]'); console.log(`Total: ${allElements.length} 要素\n`); const nodes = []; const clusters = []; allElements.forEach(el => { const id = el.getAttribute('data-id'); const et = el.getAttribute('data-et'); if (et === 'node') { nodes.push(id); } else if (et === 'cluster') { clusters.push(id); } }); console.log('📦 個別ノード [data-et="node"]:'); nodes.forEach((id, i) => { console.log(` ${i + 1}. ${id}`); }); console.log(`\n📦 サブグラフ/クラスタ [data-et="cluster"]:`); clusters.forEach((id, i) => { console.log(` ${i + 1}. ${id}`); }); })();
コマンド2: "DAT" を含む ID を検索
目的: DAT_04 に類似した ID を探す
(function() { const svg = document.querySelector('.svg-content-wrapper svg') || document.querySelector('svg'); console.log('=== "DAT" を含む data-id を検索 ==='); const allElements = svg.querySelectorAll('[data-id]'); const matchingElements = []; allElements.forEach(el => { const id = el.getAttribute('data-id'); if (id && id.includes('DAT')) { matchingElements.push({ id: id, et: el.getAttribute('data-et'), bbox: el.getBBox() }); } }); if (matchingElements.length > 0) { console.log(`✅ ${matchingElements.length} 件見つかりました:`); matchingElements.forEach((item, i) => { console.log(`\n${i + 1}. ${item.id}`); console.log(` data-et: ${item.et}`); console.log(` BBox: (${item.bbox.x.toFixed(1)}, ${item.bbox.y.toFixed(1)}) ${item.bbox.width.toFixed(1)}x${item.bbox.height.toFixed(1)}`); }); } else { console.log('❌ "DAT" を含む data-id が見つかりません'); } })();
コマンド3: クリック要素の ID を取得（インタラクティブ）
手順:
  1. このコマンドを実行
  2. フローチャート上の任意の要素をクリック
  3. その要素の data-id と data-et がコンソールに表示されます
  4. DAT_04 だと思っている要素をクリックして、実際の ID を確認してください
(function() { const svg = document.querySelector('.svg-content-wrapper svg') || document.querySelector('svg'); console.log('=== クリック要素検出モード有効化 ==='); console.log('フローチャート上の任意の要素をクリックしてください'); // 既存のリスナーを削除（重複防止） if (window._debugClickListener) { svg.removeEventListener('click', window._debugClickListener); } window._debugClickListener = function(e) { // data-id を持つ最も近い祖先要素を探す let target = e.target; let foundElement = null; while (target && target !== svg) { if (target.hasAttribute('data-id')) { foundElement = target; break; } target = target.parentElement; } if (foundElement) { const id = foundElement.getAttribute('data-id'); const et = foundElement.getAttribute('data-et'); const bbox = foundElement.getBBox(); console.log('\n🎯 クリックした要素:'); console.log(` data-id: ${id}`); console.log(` data-et: ${et}`); console.log(` BBox:`, { x: bbox.x, y: bbox.y, width: bbox.width, height: bbox.height, centerX: bbox.x + bbox.width / 2, centerY: bbox.y + bbox.height / 2 }); console.log(` クリック座標 (screen): (${e.clientX}, ${e.clientY})`); // SVG 座標に変換 const pt = svg.createSVGPoint(); pt.x = e.clientX; pt.y = e.clientY; const svgPt = pt.matrixTransform(svg.getScreenCTM().inverse()); console.log(` クリック座標 (SVG): (${svgPt.x.toFixed(1)}, ${svgPt.y.toFixed(1)})`); } else { console.log('⚠️ data-id を持つ要素が見つかりません'); } }; svg.addEventListener('click', window._debugClickListener); console.log('✅ 準備完了。要素をクリックしてください。'); })();
