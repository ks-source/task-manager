(function() { console.log('=== 距離計算監視モードを有効化 ==='); console.log('「✏️ エッジ追加」をクリックして、DAT_04 をクリックしてください'); // calcBoundaryDistance のテスト版を作成 window.testCalcBoundaryDistance = function(x, y, bbox, elementName) { const left = bbox.x; const right = bbox.x + bbox.width; const top = bbox.y; const bottom = bbox.y + bbox.height; console.log(`\n🧮 [${elementName}] 距離計算:`); console.log(` BBox: (${left.toFixed(1)}, ${top.toFixed(1)}) → (${right.toFixed(1)}, ${bottom.toFixed(1)})`); console.log(` Click: (${x.toFixed(1)}, ${y.toFixed(1)})`); // Inside bbox check if (x >= left && x <= right && y >= top && y <= bottom) { console.log(` 結果: INSIDE (距離 = 0)`); return 0; } // Calculate distance to nearest edge let dx = 0; let dy = 0; if (x < left) { dx = left - x; console.log(` dx = ${dx.toFixed(1)} (click は左側)`); } else if (x > right) { dx = x - right; console.log(` dx = ${dx.toFixed(1)} (click は右側)`); } else { console.log(` dx = 0 (click は水平範囲内)`); } if (y < top) { dy = top - y; console.log(` dy = ${dy.toFixed(1)} (click は上側)`); } else if (y > bottom) { dy = y - bottom; console.log(` dy = ${dy.toFixed(1)} (click は下側)`); } else { console.log(` dy = 0 (click は垂直範囲内)`); } const distance = Math.sqrt(dx * dx + dy * dy); console.log(` 結果: 距離 = ${distance.toFixed(1)}px`); return distance; }; console.log('✅ 準備完了。DAT_04 をクリックしてください。'); })();
