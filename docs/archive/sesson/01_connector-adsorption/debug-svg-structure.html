🔍 SVG構造確認コマンド

SVGの子要素構造を確認（edgePaths を探す）
目的: `<g class="edgePaths">` が存在するか、その親要素を特定します
(function() { const svg = document.querySelector('.svg-content-wrapper svg') || document.querySelector('svg'); console.log('=== SVG直下の子要素 ==='); Array.from(svg.children).forEach((child, index) => { console.log(`${index}: <${child.tagName} class="${child.className?.baseVal || child.className}">`); }); console.log('\n=== edgePaths グループの検索 ==='); const edgePaths = svg.querySelector('g.edgePaths'); if (edgePaths) { console.log('✅ edgePaths found!'); console.log(' Parent:', edgePaths.parentElement?.tagName, edgePaths.parentElement?.className?.baseVal); console.log(' Parent の親:', edgePaths.parentElement?.parentElement?.tagName); } else { console.log('❌ edgePaths not found'); } console.log('\n=== root グループの検索 ==='); const root = svg.querySelector('g.root'); if (root) { console.log('✅ root found!'); console.log(' Parent:', root.parentElement?.tagName, root.parentElement?.className?.baseVal); console.log(' root の子要素:'); Array.from(root.children).forEach((child, index) => { console.log(` ${index}: <${child.tagName} class="${child.className?.baseVal || child.className}">`); }); } else { console.log('❌ root not found'); } })();
