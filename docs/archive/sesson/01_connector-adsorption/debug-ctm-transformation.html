(function() { const svg = document.querySelector('.svg-content-wrapper svg') || document.querySelector('svg'); const scopeOk = svg.querySelector('[data-id="SCOPE_OK"]'); if (!scopeOk) { console.log('❌ SCOPE_OK not found'); コマンド2: クリック位置 (1350, 2152) との距離計算
目的: 最後にクリックした位置から SCOPE_OK と PHASE2 への距離を比較
(function() { const svg = document.querySelector('.svg-content-wrapper svg') || document.querySelector('svg'); const scopeOk = svg.querySelector('[data-id="SCOPE_OK"]'); const phase2 = svg.querySelector('[data-id="PHASE2"]'); const clickX = 1350.0; const clickY = 2152.8; console.log('=== 距離計算比較 ==='); console.log(`クリック位置: (${clickX}, ${clickY})`); const calcBoundaryDistance = (x, y, bbox) => { const left = bbox.x; const right = bbox.x + bbox.width; const top = bbox.y; const bottom = bbox.y + bbox.height; if (x >= left && x <= right && y >= top && y <= bottom) { return 0; } let dx = 0; let dy = 0; if (x < left) { dx = left - x; } else if (x > right) { dx = x - right; } if (y < top) { dy = top - y; } else if (y > bottom) { dy = y - bottom; } return Math.sqrt(dx * dx + dy * dy); }; const getBBoxInSVGCoordinates = (element) => { const bbox = element.getBBox(); const ctm = element.getCTM(); if (!ctm) { return { x: bbox.x, y: bbox.y, width: bbox.width, height: bbox.height }; } const transformPoint = (x, y, ctm) => { return { x: ctm.a * x + ctm.c * y + ctm.e, y: ctm.b * x + ctm.d * y + ctm.f }; }; const topLeft = transformPoint(bbox.x, bbox.y, ctm); const topRight = transformPoint(bbox.x + bbox.width, bbox.y, ctm); const bottomLeft = transformPoint(bbox.x, bbox.y + bbox.height, ctm); const bottomRight = transformPoint(bbox.x + bbox.width, bbox.y + bbox.height, ctm); const minX = Math.min(topLeft.x, topRight.x, bottomLeft.x, bottomRight.x); const maxX = Math.max(topLeft.x, topRight.x, bottomLeft.x, bottomRight.x); const minY = Math.min(topLeft.y, topRight.y, bottomLeft.y, bottomRight.y); const maxY = Math.max(topLeft.y, topRight.y, bottomLeft.y, bottomRight.y); return { x: minX, y: minY, width: maxX - minX, height: maxY - minY }; }; if (scopeOk) { const bboxScopeOk = getBBoxInSVGCoordinates(scopeOk); const distScopeOk = calcBoundaryDistance(clickX, clickY, bboxScopeOk); console.log('\n📦 SCOPE_OK:'); console.log(' BBox:', bboxScopeOk); console.log(' 境界距離:', distScopeOk.toFixed(1), 'px'); } if (phase2) { const bboxPhase2 = getBBoxInSVGCoordinates(phase2); const distPhase2 = calcBoundaryDistance(clickX, clickY, bboxPhase2); console.log('\n📦 PHASE2:'); console.log(' BBox:', bboxPhase2); console.log(' 境界距離:', distPhase2.toFixed(1), 'px'); } console.log('\n🎯 結果:'); if (scopeOk && phase2) { const bboxScopeOk = getBBoxInSVGCoordinates(scopeOk); const bboxPhase2 = getBBoxInSVGCoordinates(phase2); const distScopeOk = calcBoundaryDistance(clickX, clickY, bboxScopeOk); const distPhase2 = calcBoundaryDistance(clickX, clickY, bboxPhase2); if (distScopeOk < distPhase2) { console.log('✅ SCOPE_OK の方が近い'); } else { console.log('❌ PHASE2 の方が近い（または同じ）'); } } })();
コマンド3: 全個別ノードの変換後 BBox リスト
目的: すべての個別ノード（data-et="node"）の変換後 BBox を確認
(function() { const svg = document.querySelector('.svg-content-wrapper svg') || document.querySelector('svg'); const nodes = svg.querySelectorAll('[data-et="node"]'); const getBBoxInSVGCoordinates = (element) => { const bbox = element.getBBox(); const ctm = element.getCTM(); if (!ctm) { return { x: bbox.x, y: bbox.y, width: bbox.width, height: bbox.height }; } const transformPoint = (x, y, ctm) => { return { x: ctm.a * x + ctm.c * y + ctm.e, y: ctm.b * x + ctm.d * y + ctm.f }; }; const topLeft = transformPoint(bbox.x, bbox.y, ctm); const topRight = transformPoint(bbox.x + bbox.width, bbox.y, ctm); const bottomLeft = transformPoint(bbox.x, bbox.y + bbox.height, ctm); const bottomRight = transformPoint(bbox.x + bbox.width, bbox.y + bbox.height, ctm); const minX = Math.min(topLeft.x, topRight.x, bottomLeft.x, bottomRight.x); const maxX = Math.max(topLeft.x, topRight.x, bottomLeft.x, bottomRight.x); const minY = Math.min(topLeft.y, topRight.y, bottomLeft.y, bottomRight.y); const maxY = Math.max(topLeft.y, topRight.y, bottomLeft.y, bottomRight.y); return { x: minX, y: minY, width: maxX - minX, height: maxY - minY }; }; console.log('=== 全個別ノードの変換後 BBox ==='); console.log(`Total: ${nodes.length} nodes\n`); nodes.forEach((node, i) => { const id = node.getAttribute('data-id'); const bbox = getBBoxInSVGCoordinates(node); console.log(`${i + 1}. ${id}:`); console.log(` BBox: (${bbox.x.toFixed(1)}, ${bbox.y.toFixed(1)}) ${bbox.width.toFixed(1)}x${bbox.height.toFixed(1)}`); }); })();
